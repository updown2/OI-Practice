#include<bits/stdc++.h>
using namespace std;
#define For(i, a, b) for(int i=a; i<b; i++)
#define ffi For(i, 0, N)
#define ffj For(j, 0, M)
#define ffa ffi ffj
#define int long long
#define mp make_pair
#define a first
#define b second
#define w cout
#define e "\n"
#define s <<" "<<
///////////////////////////////////////////
const int MAXN = 10;
/// 500,000,000
int N, M, grid[MAXN][MAXN], out;
bool vis[MAXN][MAXN];
priority_queue<pair<int, pair<int, int> > > next1;
 
main () {
    //ifstream cin("test.in");
    ios_base::sync_with_stdio(0); cin.tie(0);
    cin >> N >> M;
    ffa {
        cin >> grid[i][j];
        next1.push(mp(grid[i][j], mp(i, j)));
    }
    while (!next1.empty()) {
        int i = next1.top().b.a, j = next1.top().b.b;
        next1.pop();
        out += grid[i][j];
        if (i != 0 && vis[i-1][j]) out += grid[i-1][j];
        if (i != N-1 && vis[i+1][j]) out += grid[i+1][j];
        if (j != 0 && vis[i][j-1]) out += grid[i][j-1];
        if (j != M-1 && vis[i][j+1]) out += grid[i][j+1];
        vis[i][j] = true;
    }
    w<< out<<e;
}
