int N;
vector<int> adj[100000];
bool vis[100000], vis2[100000];
int best;
int node;
int out;
 
void go(int at, int dep) {
    if (vis[at]) return;
    vis[at] = true;
    if (dep > best) best=dep, node=at;
    for (int i: adj[at]) go(i, dep+1);
}
 
void go2(int at, int dep) {
    if (vis2[at]) return;
    vis2[at] = true;
    out = max(out, dep);
    for (int i: adj[at]) go2(i, dep+1);
}

        int a, b; cin >> a >> b; a--; b--;
        adj[a].pb(b); adj[b].pb(a);
    }
    go(0, 0);
    go2(node, 0);
    w<<out<<e;
}
