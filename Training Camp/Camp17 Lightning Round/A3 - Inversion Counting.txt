int N, out, vals[MAXN];

void solve(int a, int b) {
    //w<< a s b<<e;
    if (a == b) return;
    solve(a, (a+b)/2), solve((a+b)/2+1, b);
    double use[b-a+1];
    int at1 = a, at2 = (a+b)/2+1;
    For (i, 0, b-a+1) {
        if (at1 == (a+b)/2+1) {
            /// can't use at1
            use[i] = vals[at2];
            at2++;
        }
        else if (at2 == b+1) {
            /// can't use at2
            use[i] = vals[at1];
            at1++;
        }
        else {
            /// can use at1 and at2
            if (vals[at1] <= vals[at2]) {
                use[i] = vals[at1];
                at1++;
            }
            else {
                use[i] = vals[at2];
                at2++;
                out += (a+b)/2+1 - at1;
            }
        }
    }
    For (i, 0, b-a+1) vals[i+a] = use[i];
}
