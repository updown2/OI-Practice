ll out = 0;

ll sear(ll val) {
    ll small = 0;
    ll big = N;
    while (small != big) {
        ll mid = (small+big+1)/2;
        if (len[mid] < val) small = mid;
        else big = mid-1;
    }
    return small+1;
}

        small = min(small, inp[i]);
    }
    For (i, 0, N+1) len[i] = big+1;
    len[0] = small-1;
 
	ffi {

}
