const int MAXN = 100000, INF=10000000, MOD = 1000000007;
////////////////////////////////////////////////////////
int N, par[MAXN], num = INF, siz[MAXN], bund[MAXN];
bool vis[MAXN];
vector<int> inp[MAXN], adj[MAXN], out;

void go(int at, int p) {
    vis[at] = true;
    par[at] = p;
    siz[at] = 1;
    for (int i: inp[at]) if (!vis[i]) {
        adj[at].pb(i);
        go(i, at);
    }
    /// check
    int val = max(bund[at], N-siz[at]);
    if (val < num) {
        num = val;
        out.clear();
    }
    if (val == num) out.pb(at+1);
    //w<< at+1 s bund[at] s N-siz[at]<<e;
    if (p == -1) return;
    /// update parent
    bund[p] = max(bund[p], siz[at]);
    siz[p] += siz[at];
}

main() {
	//ifstream cin ("test.in");
	ios_base::sync_with_stdio(0); cin.tie(0);
	cin >> N;
	For (i, 1, N) {
        int a, b; cin >> a >> b; a--; b--;
        inp[a].pb(b), inp[b].pb(a);
	}
	go(0, -1);
	//w<< num<<e;
	sort(out.begin(), out.end());
	for (int i: out) w<< i<< " "; w<<e;
}
