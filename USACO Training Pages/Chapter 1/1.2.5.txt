#include <iostream>

using namespace std;
//Global Variables
int base;

//useful functions
string reverseChar(string str) {
	string out;
	for (int i=str.length()-1; i > -1; i--) {
		out += str[i];
	}
	return out;
}

bool is_pal(string check) {
	bool out = true;
	for (int i=0; i < check.length(); i++) {
		if (check[i] != check[check.length() - (i+1)]) {
			out = false;
		}
	}
	return out;
}
